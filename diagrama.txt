@startuml DiagramaZeroWasteAlchemist

class Person {
  -first_name: Char
  -last_name: Char
  -email: Email
  -phone: Char
  +toString()
}

class Player {
  -username: Char
  -level: Integer
  -current_xp: Integer
  -gold_balance: Decimal
  -reputation_score: Integer
  +calculateLevel()
}

class Category {
  -name: Char
  -description: Text
  -storage_tips: Text
  -icon_identifier: Char
  +toString()
}

class Ingredient {
  -name: Char
  -brand: Text
  -expiration_date: Date
  -quantity: Decimal
  -unit_of_measure: Char
  -is_organic: Boolean
  -category: ForeignKey[Category]
  -location: ForeignKey[StorageLocation]
  +checkFreshness()
}

class StorageLocation {
  -name: Char
  -type: CharChoices
  -temperature_setting: Decimal
  -capacity_liters: Decimal
  +toString()
}

class Recipe {
  -title: Char
  -instructions: Text
  -difficulty: Integer
  -prep_time_minutes: Integer
  -xp_reward: Integer
  +toString()
}

class IngredientRequirement {
  -required_quantity: Decimal
  -unit: Char
  -recipe: ForeignKey[Recipe]
  -ingredient_name: Char
  +checkAvailability()
}

class Achievement {
  -title: Char
  -description: Text
  -rarity_level: CharChoices
  -xp_bonus: Integer
  +toString()
}

class Mission {
  -title: Char
  -objective: Text
  -deadline: DateTime
  -reward_gold: Integer
  -is_completed: Boolean
  +toString()
}

class AlchemyLog {
  -activity_date: DateTime
  -action_type: CharChoices
  -points_gained: Integer
  -waste_saved_grams: Decimal
  -player: ForeignKey[Player]
  +toString()
}

class PlayerAchievement {
  -unlock_date: Date
  -progress_percentage: Decimal
  -player: ForeignKey[Player]
  -achievement: ForeignKey[Achievement]
  +toString()
}

' Relacionamentos e Herança
Player -up-|> Person : 1..1
Ingredient "n" -- "1" Category : pertence a
Ingredient "n" -- "1" StorageLocation : armazenado em
Player "1" *-- "n" Ingredient : possui
Recipe "1" *-- "n" IngredientRequirement : exige
Player "1" *-- "n" AlchemyLog : registra
Player "1" *-- "n" Mission : participa
Player "1" *-- "n" PlayerAchievement : desbloqueia
Achievement "1" *-- "n" PlayerAchievement : concedida em

@enduml
